# React Hook Form Basics

## Learning Objectives

By the end of this lesson, you will:

-  [ ] Install and configure React Hook Form in TaskFlow Pro.

## Basic Form Example

```tsx
import { useForm } from "react-hook-form";
import { zodResolver } from "@hookform/resolvers/zod";

const signUpSchema = z.object({
   name: z.string().min(2),
   email: z.string().email(),
   password: z.string().min(8),
});

type SignUpInput = z.infer<typeof signUpSchema>;

export function SignUpForm() {
   const {
      register,
      handleSubmit,
      formState: { errors, isSubmitting },
   } = useForm<SignUpInput>({ resolver: zodResolver(signUpSchema) });

   const onSubmit = async (values: SignUpInput) => {
      await authClient.signUp(values);
   };

   return (
      <form onSubmit={handleSubmit(onSubmit)} className='space-y-4'>
         <input type='text' placeholder='Name' {...register("name")} />
         {errors.name && <p role='alert'>{errors.name.message}</p>}

         <input type='email' placeholder='Email' {...register("email")} />
         {errors.email && <p role='alert'>{errors.email.message}</p>}

         <input
            type='password'
            placeholder='Password'
            {...register("password")}
         />
         {errors.password && <p role='alert'>{errors.password.message}</p>}

         <button type='submit' disabled={isSubmitting}>
            Create account
         </button>
      </form>
   );
}
```

## Persisting Draft Data

-  Use `useForm`'s `defaultValues` with Zustand or Context to hydrate drafts.
-  Sync to local storage when steps span multiple pages.

<Accordions type='single' className='mt-4'>
   <Accordion title='TaskFlow Pro Setup'>
      <p>Wire React Hook Form into the project scaffold.</p>

      ```tsx filename="src/forms/useSignUpForm.ts"
      import { useForm } from "react-hook-form";
      import { zodResolver } from "@hookform/resolvers/zod";
      import { signUpSchema } from "@/validation/schemas";

      export function useSignUpForm(defaultValues?: Partial<SignUpInput>) {
         return useForm<SignUpInput>({
            resolver: zodResolver(signUpSchema),
            defaultValues,
         });
      }
      ```
