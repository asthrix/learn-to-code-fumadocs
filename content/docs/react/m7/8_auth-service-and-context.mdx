# Auth Service & Context

## Axios Instance

```ts filename="src/lib/http.ts"
import axios from "axios";

export const http = axios.create({
   baseURL: process.env.NEXT_PUBLIC_API_URL,
   withCredentials: true,
});

http.interceptors.request.use((config) => {
   const token = sessionStorage.getItem("taskflow-access-token");
   if (token) {
      config.headers.Authorization = `Bearer ${token}`;
   }
   return config;
});
```

## Auth Service

```ts filename="src/lib/auth-service.ts"
import { http } from "./http";

type LoginPayload = { email: string; password: string };
type LoginResponse = { accessToken: string };

export const AuthService = {
   login: async (payload: LoginPayload) => {
      const { data } = await http.post<LoginResponse>("/login", payload);
      sessionStorage.setItem("taskflow-access-token", data.accessToken);
      return data;
   },
   logout: async () => {
      await http.post("/logout");
      sessionStorage.removeItem("taskflow-access-token");
   },
   refresh: async () => {
      const { data } = await http.post<LoginResponse>("/refresh");
      sessionStorage.setItem("taskflow-access-token", data.accessToken);
      return data;
   },
};
```

## Session Store

```ts filename="src/lib/session-store.ts"
import { create } from "zustand";

interface SessionState {
   accessToken: string | null;
   setAccessToken: (token: string | null) => void;
}

export const useSessionStore = create<SessionState>((set) => ({
   accessToken: null,
   setAccessToken: (token) => set({ accessToken: token }),
}));

export const hydrateSession = () => {
   const token = sessionStorage.getItem("taskflow-access-token");
   useSessionStore.getState().setAccessToken(token);
};
```

## Auth Provider & Hooks

```tsx filename="src/components/auth/AuthProvider.tsx"
import { ReactNode, useEffect, useState } from "react";
import { hydrateSession, useSessionStore } from "@/lib/session-store";

export const AuthContext = React.createContext({
   accessToken: null as string | null,
   isLoading: true,
});

export const AuthProvider = ({ children }: { children: ReactNode }) => {
   const [isLoading, setIsLoading] = useState(true);
   const accessToken = useSessionStore((state) => state.accessToken);

   useEffect(() => {
      hydrateSession();
      setIsLoading(false);
   }, []);

   return (
      <AuthContext.Provider value={{ accessToken, isLoading }}>
         {children}
      </AuthContext.Provider>
   );
};
```

```ts filename="src/hooks/useAuth.ts"
import { useContext } from "react";
import { AuthContext } from "@/components/auth/AuthProvider";

export const useAuth = () => useContext(AuthContext);

export const useIsAuthenticated = () => {
   const { accessToken, isLoading } = useAuth();
   return { isAuthenticated: Boolean(accessToken), isLoading };
};
```

## Route Guard

```tsx filename="src/components/auth/ProtectedRoute.tsx"
import { ReactNode, useEffect } from "react";
import { useRouter } from "next/navigation";
import { useIsAuthenticated } from "@/hooks/useAuth";

export const ProtectedRoute = ({ children }: { children: ReactNode }) => {
   const router = useRouter();
   const { isAuthenticated, isLoading } = useIsAuthenticated();

   useEffect(() => {
      if (!isLoading && !isAuthenticated) {
         router.replace(
            `/login?redirect=${encodeURIComponent(window.location.pathname)}`
         );
      }
   }, [isAuthenticated, isLoading, router]);

   if (isLoading) return <p>Loading session…</p>;

   return <>{children}</>;
};
```
