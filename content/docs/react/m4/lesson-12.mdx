## Core Concepts Summary

### 1. React Router Setup

```jsx
// Router.jsx - Central routing configuration
import { BrowserRouter, Routes, Route } from 'react-router-dom';

export default function Router() {
  return (
    <BrowserRouter>
      <Routes>
        <Route path="/" element={<HomePage />} />
        <Route path="/listing/:id" element={<ListingDetailsPage />} />
        <Route path="*" element={<NotFoundPage />} />
      </Routes>
    </BrowserRouter>
  );
}
```

**Key components:**
- **BrowserRouter** - Provides routing context
- **Routes** - Container for route definitions
- **Route** - Maps paths to components

### 2. Dynamic Routes

```jsx
// Route with parameter
<Route path="/listing/:id" element={<ListingDetailsPage />} />

// Extract parameter in component
import { useParams } from 'react-router-dom';

function ListingDetailsPage() {
  const { id } = useParams(); // id = "42" from /listing/42
  
  useEffect(() => {
    fetch(`/api/listings/${id}`)
      .then(res => res.json())
      .then(setListing);
  }, [id]); // Re-fetch when ID changes
}
```

**Remember:**
- Parameters are always strings
- Include param in useEffect dependencies
- Use template literals for dynamic URLs

## Common Patterns

<Accordions>

<Accordion title="Pattern 1: Page with Data Fetching">

```jsx
import { useParams } from 'react-router-dom';
import { useState, useEffect } from 'react';

function DetailsPage() {
  const { id } = useParams();
  const [data, setData] = useState(null);
  const [isLoading, setIsLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    const controller = new AbortController();

    fetch(`/api/items/${id}`, { signal: controller.signal })
      .then(res => {
        if (!res.ok) throw new Error('Not found');
        return res.json();
      })
      .then(setData)
      .catch(err => {
        if (err.name !== 'AbortError') setError(err.message);
      })
      .finally(() => setIsLoading(false));

    return () => controller.abort();
  }, [id]);

  if (isLoading) return <Spinner />;
  if (error) return <ErrorMessage error={error} />;

  return <DisplayComponent data={data} />;
}
```

</Accordion>

<Accordion title="Pattern 2: Protected Route">

```jsx
function ProtectedPage() {
  const navigate = useNavigate();
  const { user } = useAuth();

  useEffect(() => {
    if (!user) {
      navigate('/login', {
        state: { from: location.pathname }
      });
    }
  }, [user, navigate]);

  return user ? <ProtectedContent /> : null;
}
```

</Accordion>

function BookingForm() {

## Best Practices

### Routing Best Practices

1. **Route Organization**
   ```jsx
   // ‚úÖ Keep routes in one central file
   // src/components/Router.jsx
   
   // ‚ùå Don't scatter routes across files
   ```

2. **Route Order**
   ```jsx
   <Routes>
     {/* Specific routes first */}
     <Route path="/listing/new" element={<CreateListing />} />
     
     {/* Dynamic routes next */}
     <Route path="/listing/:id" element={<Details />} />
     
     {/* Catch-all last */}
     <Route path="*" element={<NotFound />} />
   </Routes>
   ```

3. **URL Design**
   ```jsx
   // ‚úÖ Good - clear, RESTful
   /listings           // List all
   /listing/:id        // View one
   /listing/:id/edit   // Edit one
   
   // ‚ùå Bad - unclear
   /view?id=42
   /edit-listing-42
   ```

### Component Best Practices

1. **Separate Concerns**
   ```jsx
   // ‚úÖ Page handles data, Card handles display
   function ListingDetailsPage() {
     const { id } = useParams();
     const listing = useFetch(`/api/listings/${id}`);
     return <ListingDetailsCard listing={listing} />;
   }
   ```

2. **Error Handling**
   ```jsx
   // ‚úÖ Always handle loading and errors
   if (isLoading) return <Spinner />;
   if (error) return <ErrorMessage error={error} />;
   return <Content data={data} />;
   ```

3. **Cleanup**
   ```jsx
   // ‚úÖ Always cleanup side effects
   useEffect(() => {
     const controller = new AbortController();
     
     fetch(url, { signal: controller.signal })
       .then(/* ... */);
     
     return () => controller.abort();
   }, [id]);
   ```

## Common Pitfalls

<Accordions>

<Accordion title="Forgetting Dependencies">

```jsx
// ‚ùå Wrong - missing id dependency
useEffect(() => {
  fetch(`/api/listings/${id}`)
    .then(setListing);
}, []); // Won't re-fetch when ID changes!

// ‚úÖ Correct - include id
useEffect(() => {
  fetch(`/api/listings/${id}`)
    .then(setListing);
}, [id]); // Re-fetches when ID changes
```

</Accordion>

<Accordion title="Using <a> Instead of Link">

```jsx
// ‚ùå Wrong - full page reload
<a href="/listing/42">View</a>

// ‚úÖ Correct - client-side navigation
<Link to="/listing/42">View</Link>
```

</Accordion>

<Accordion title="Catch-All Route Order">

```jsx
// ‚ùå Wrong - catch-all first
<Routes>
  <Route path="*" element={<NotFound />} />
  <Route path="/listing/:id" element={<Details />} />
</Routes>

// ‚úÖ Correct - catch-all last
<Routes>
  <Route path="/listing/:id" element={<Details />} />
  <Route path="*" element={<NotFound />} />
</Routes>
```

</Accordion>

<Accordion title="Forgetting Route Cleanup">

```jsx
// ‚ùå Wrong - no cleanup
useEffect(() => {
  fetch(`/api/listings/${id}`)
    .then(setListing);
}, [id]); // Race conditions possible!

// ‚úÖ Correct - with cleanup
useEffect(() => {
  const controller = new AbortController();
  
  fetch(`/api/listings/${id}`, { 
    signal: controller.signal 
  })
    .then(setListing);
    
  return () => controller.abort();
}, [id]);
```

</Accordion>

## React Router Hooks Quick Reference

| Hook | Purpose | Example |
|------|---------|---------|
| `useParams` | Get URL parameters | `const { id } = useParams()` |
| `useNavigate` | Programmatic navigation | `navigate('/home')` |
| `useLocation` | Get current location | `const location = useLocation()` |
| `useSearchParams` | Query string params | `const [params] = useSearchParams()` |
| `useMatch` | Check route match | `const match = useMatch('/listing/:id')` |
