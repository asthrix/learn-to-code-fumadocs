## Phase 2: Tailwind CSS Setup

### Step 1: Install Tailwind

```bash
# Install Tailwind and dependencies
npm install -D tailwindcss postcss autoprefixer

# Generate config files
npx tailwindcss init -p
```

### Step 2: Configure Tailwind

Update `tailwind.config.js`:

```javascript
/** @type {import('tailwindcss').Config} */
export default {
  content: [
    "./index.html",
    "./src/**/*.{js,ts,jsx,tsx}",
  ],
  theme: {
    extend: {
      colors: {
        primary: {
          50: '#eff6ff',
          100: '#dbeafe',
          200: '#bfdbfe',
          300: '#93c5fd',
          400: '#60a5fa',
          500: '#3b82f6',
          600: '#2563eb',
          700: '#1d4ed8',
          800: '#1e40af',
          900: '#1e3a8a',
        },
      },
    },
  },
  plugins: [],
}
```

### Step 3: Add Tailwind Directives

Replace `src/index.css` contents:

```css
@tailwind base;
@tailwind components;
@tailwind utilities;

/* Custom base styles */
@layer base {
  body {
    @apply bg-gray-50 text-gray-900;
  }
}

/* Custom component styles */
@layer components {
  .btn {
    @apply px-4 py-2 rounded-lg font-medium transition-colors;
  }
  
  .btn-primary {
    @apply btn bg-primary-600 text-white hover:bg-primary-700;
  }
  
  .btn-secondary {
    @apply btn bg-gray-200 text-gray-900 hover:bg-gray-300;
  }
  
  .card {
    @apply bg-white rounded-lg shadow-sm border border-gray-200 p-6;
  }
}
```

## Phase 3: Code Quality Tools

### Step 1: Install ESLint and Prettier

```bash
# Install ESLint
npm install -D eslint eslint-plugin-react eslint-plugin-react-hooks eslint-plugin-react-refresh

# Install Prettier
npm install -D prettier eslint-config-prettier eslint-plugin-prettier
```

### Step 2: Configure ESLint

Create `.eslintrc.json`:

```json
{
  "env": {
    "browser": true,
    "es2021": true,
    "node": true
  },
  "extends": [
    "eslint:recommended",
    "plugin:react/recommended",
    "plugin:react/jsx-runtime",
    "plugin:react-hooks/recommended",
    "prettier"
  ],
  "parserOptions": {
    "ecmaVersion": "latest",
    "sourceType": "module",
    "ecmaFeatures": {
      "jsx": true
    }
  },
  "plugins": [
    "react",
    "react-hooks",
    "react-refresh",
    "prettier"
  ],
  "rules": {
    "react/prop-types": "off",
    "react-refresh/only-export-components": "warn",
    "prettier/prettier": "error",
    "no-console": "warn",
    "no-unused-vars": "warn"
  },
  "settings": {
    "react": {
      "version": "detect"
    }
  }
}
```

### Step 3: Configure Prettier

Create `.prettierrc`:

```json
{
  "semi": true,
  "singleQuote": true,
  "tabWidth": 2,
```


Create `.env.example`:

```bash

# App Configuration
VITE_APP_NAME=TaskFlow Pro
VITE_APP_VERSION=0.1.0

# API Configuration
VITE_API_URL=http://localhost:3001/api
VITE_API_KEY=your-api-key-here
```

### Step 2: Create Configuration Module

Create `src/config/env.js`:

```javascript
/**
 * Application configuration
 * Centralizes all environment variables with defaults
 */

export const config = {
  app: {
    name: import.meta.env.VITE_APP_NAME || 'TaskFlow Pro',
    version: import.meta.env.VITE_APP_VERSION || '0.1.0',
    description: import.meta.env.VITE_APP_DESCRIPTION || '',
    mode: import.meta.env.MODE,
    isDev: import.meta.env.DEV,
    isProd: import.meta.env.PROD,
  },

  api: {
    url: import.meta.env.VITE_API_URL || 'http://localhost:3001/api',
    timeout: parseInt(import.meta.env.VITE_API_TIMEOUT || '5000', 10),
    key: import.meta.env.VITE_API_KEY || '',
  },

  features: {
    debug: import.meta.env.VITE_ENABLE_DEBUG === 'true',
    devTools: import.meta.env.VITE_SHOW_DEV_TOOLS === 'true',
  },
};

// Log configuration in development
if (config.app.isDev) {
  console.log('🚀 TaskFlow Pro Configuration:', {
    mode: config.app.mode,
    version: config.app.version,
    apiUrl: config.api.url,
    debug: config.features.debug,
  });
}

export default config;
```

### Step 3: Create UI Components

Create `src/components/ui/Card.jsx`:

```jsx
function Card({ title, description, status, priority }) {
  const statusColors = {
    pending: 'bg-yellow-100 text-yellow-800',
    'in-progress': 'bg-blue-100 text-blue-800',
    completed: 'bg-green-100 text-green-800',
  };

  const priorityIcons = {
    low: '🔵',
    medium: '🟡',
    high: '🔴',
  };

  return (
    <div className="card hover:shadow-md transition-shadow">
      <div className="flex items-start justify-between mb-3">
        <h3 className="text-lg font-semibold text-gray-900">{title}</h3>
        <span className="text-2xl">{priorityIcons[priority]}</span>
      </div>

      <p className="text-gray-600 text-sm mb-4">{description}</p>

      <div className="flex items-center justify-between">
        <span
          className={`
            px-2 py-1 text-xs font-medium rounded-full
            ${statusColors[status]}
          `}
        >
          {status.replace('-', ' ')}
        </span>

        <div className="flex gap-2">
          <button className="text-gray-400 hover:text-primary-600 transition-colors">
            ✏️
          </button>
          <button className="text-gray-400 hover:text-red-600 transition-colors">
            🗑️
          </button>
        </div>
      </div>

  );
}

export default Card;
```

## Troubleshooting Common Issues

### Issue: Tailwind Styles Not Working

```bash
# Check tailwind.config.js content paths
# Should include: "./src/**/*.{js,jsx}"

# Clear Vite cache
rm -rf node_modules/.vite

# Restart dev server
npm run dev
```

### Issue: ESLint Errors

```bash
# Check .eslintrc.json exists and is valid JSON

# Install missing dependencies
npm install -D eslint-plugin-react eslint-plugin-react-hooks

# Restart VS Code
# Cmd+Shift+P → Reload Window
```

### Issue: Environment Variables Not Loading

```bash
# Restart dev server (required for .env changes)
# Ctrl+C
npm run dev

# Check variable names have VITE_ prefix
# VITE_API_URL ✅
# API_URL ❌
```
